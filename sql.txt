CREATE TABLE track
(
  trackID INT NOT NULL,
  name VARCHAR(10000) NOT NULL,
  date DATE NOT NULL,
  description VARCHAR(10000) NOT NULL,
  PRIMARY KEY (trackID)
);

CREATE TABLE data
(
  img VARCHAR(10000) NOT NULL,
  description VARCHAR(10000) NOT NULL,
  gameSrc VARCHAR(10000) NOT NULL,
  trackID INT NOT NULL,
  FOREIGN KEY (trackID) REFERENCES track(trackID)
);
