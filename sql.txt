Create Database codeValley;
use codeValley;
CREATE TABLE track
(
  trackID INT NOT NULL AUTO_INCREMENT,
  name TEXT NOT NULL,
  date DATE NOT NULL,
  description TEXT,
  PRIMARY KEY (trackID)
);

CREATE TABLE data
(
  img TEXT NOT NULL,
  description TEXT,
  gameSrc TEXT NOT NULL,
  trackID INT NOT NULL,
  FOREIGN KEY (trackID) REFERENCES track(trackID)
);
